# Docker Desktop
Download docker desktop
Run and open docker desktop

# Set up directory environment
mkdir -p dbt
cd dbt
touch docker-compose.yml .env profiles.yml

# Set up docker-compose.yml
services:
  postgres:
    image: postgres:15
    env_file:
      - .env
    ports:
      - "5432:5432"
  dbt:
    image: ghcr.io/dbt-labs/dbt-postgres:1.8.latest
    env_file:
      - .env
    volumes:
      - ./project/my_project:/usr/app/dbt
      - ./profiles.yml:/root/.dbt/profiles.yml
    working_dir: /usr/app/dbt
    depends_on:
      - postgres
    entrypoint: ["tail", "-f", "/dev/null"]

# Set up .env
POSTGRES_USER=[user]
POSTGRES_PASSWORD=[password]
POSTGRES_DB=[database]

# Set up profiles.yml
my_project:
  target: dev
  outputs:
    dev:
      type: postgres
      host: host.docker.internal
      user: "{{ env_var('POSTGRES_USER') }}"
      password: "{{ env_var('POSTGRES_PASSWORD') }}"
      port: 5432
      dbname: "{{ env_var('POSTGRES_DB') }}"
      schema: bronze
      threads: 1

# Run Container
docker compose up -d

# initalize
docker compose exec dbt bash
dbt init my_project --skip-profile-setup

# remove the example models
rm -rf project/my_project/models/example
touch project/my_project/models/.gitkeep
